date1 = dt.datetime(2019, 1, 1)

def daterange(d1, d2):
  return (d1 + dt.timedelta(days=i) for i in range((d2 - d1).days + 1))

for ((index,row), d) in zip(df.iterrows(), daterange(date1, date2)):
